212  procedure REPORT_MESSAGE(SELF:inout VOCAL_TYPE; TAG:inout LINE; MESSAGE:in STRING) is
213  variable text_line : LINE;
214  begin
215  if (SELF.tag_field_width > 0) then
216  WRITE(text_line, TAG(TAG'range), RIGHT, SELF.tag_field_width);
217  WRITE(text_line, string'(" "));
218  elsif (SELF.tag_field_width < 0) then
219  WRITE(text_line, TAG(TAG'range), LEFT , -SELF.tag_field_width);
220  WRITE(text_line, string'(" "));
221  end if;
222  if (SELF.time_field_width > 0) then
223  WRITE(text_line, Now, RIGHT, SELF.time_field_width);
224  elsif (SELF.time_field_width < 0) then
225  WRITE(text_line, Now, LEFT , -SELF.time_field_width);
226  end if;
227  WRITE(text_line, " (" & SELF.name(SELF.name'range) & ") " & MESSAGE);
228  WRITELINE(OUTPUT, text_line);
229  end procedure;
