409  procedure map_read_axi4_stream_signals(
410  signals : inout AXI4_STREAM_SIGNAL_TYPE;
411  event : inout EVENT_TYPE
412  ) is
413  constant proc_name : string := "MAP_READ_AXI4_STREAM_SIGNALS";
414  variable next_event : EVENT_TYPE;
415  variable key_word : KEYWORD_TYPE;
416  begin
417  REPORT_DEBUG(core, proc_name, "BEGIN");
418  next_event := event;
419  MAP_LOOP: loop
420  case next_event is
421  when EVENT_SCALAR =>
422  COPY_KEY_WORD(core, key_word);
423  case key_word is
424  when KEY_TDATA | KEY_DATA =>
425  read_value(proc_name, signals.DATA(WIDTH.DATA -1 downto 0));
426  when KEY_TKEEP | KEY_KEEP =>
427  read_value(proc_name, signals.KEEP(WIDTH.DATA/8-1 downto 0));
428  when KEY_TSTRB | KEY_STRB =>
429  read_value(proc_name, signals.STRB(WIDTH.DATA/8-1 downto 0));
430  when KEY_TUSER | KEY_USER =>
431  read_value(proc_name, signals.USER(WIDTH.USER -1 downto 0));
432  when KEY_TDEST | KEY_DEST =>
433  read_value(proc_name, signals.DEST(WIDTH.DEST -1 downto 0));
434  when KEY_TID | KEY_ID =>
435  read_value(proc_name, signals.ID (WIDTH.ID -1 downto 0));
436  when KEY_TLAST | KEY_LAST =>
437  read_value(proc_name, signals.LAST );
438  when KEY_TVALID | KEY_VALID =>
439  read_value(proc_name, signals.VALID);
440  when KEY_TREADY | KEY_READY =>
441  read_value(proc_name, signals.READY);
442  when others => exit MAP_LOOP;
443  end case;
444  when EVENT_MAP_END => exit MAP_LOOP;
445  when others => exit MAP_LOOP;
446  end case;
447  SEEK_EVENT(core, stream, next_event);
448  if (next_event = EVENT_SCALAR) then
449  READ_EVENT(core, stream, EVENT_SCALAR);
450  end if;
451  end loop;
452  event := next_event;
453  REPORT_DEBUG(core, proc_name, "END");
454  end procedure;

509  procedure execute_wait is
510  constant proc_name : string := "EXECUTE_WAIT";
511  variable next_event : EVENT_TYPE;
512  variable keyword : KEYWORD_TYPE;
513  variable wait_count : integer;
514  variable scan_len : integer;
515  variable timeout : integer;
516  variable wait_on : boolean;
517  variable axi_match : boolean;
518  variable gpi_match : boolean;
519  begin
520  REPORT_DEBUG(core, proc_name, "BEGIN");
521  timeout := DEFAULT_WAIT_TIMEOUT;
522  wait_on := FALSE;
523  SEEK_EVENT(core, stream, next_event);
524  case next_event is
525  when EVENT_SCALAR =>
526  READ_EVENT(core, stream, EVENT_SCALAR);
527  STRING_TO_INTEGER(
528  STR => core.str_buf(1 to core.str_len),
529  VAL => wait_count,
530  STR_LEN => scan_len
531  );
532  if (scan_len = 0) then
533  wait_count := 1;
534  end if;
535  if (wait_count > 0) then
536  for i in 1 to wait_count loop
537  wait until (ACLK'event and ACLK = '1');
538  end loop;
539  end if;
540  wait_count := 0;
541  when EVENT_MAP_BEGIN =>
542  READ_EVENT(core, stream, EVENT_MAP_BEGIN);
543  chk_signals := AXI4_STREAM_SIGNAL_DONTCARE;
544  gpi_signals := (others => '-');
545  MAP_READ_LOOP: loop
546  REPORT_DEBUG(core, proc_name, "MAP_READ_LOOP");
547  MAP_READ_PREPARE_FOR_NEXT(
548  SELF => core , -- I/O:
549  STREAM => stream , -- I/O:
550  EVENT => next_event -- I/O:
551  );
552  map_read_axi4_stream_signals(
553  signals => chk_signals , -- I/O:
554  event => next_event -- I/O:
555  );
556  MAP_READ_STD_LOGIC_VECTOR(
557  SELF => core , -- I/O:
558  STREAM => stream , -- I/O:
559  KEY => "GPI" , -- In :
560  VAL => gpi_signals , -- I/O:
561  EVENT => next_event -- I/O:
562  );
563  MAP_READ_INTEGER(
564  SELF => core , -- I/O:
565  STREAM => stream , -- I/O:
566  KEY => "TIMEOUT" , -- In :
567  VAL => timeout , -- I/O:
568  EVENT => next_event -- I/O:
569  );
570  MAP_READ_BOOLEAN(
571  SELF => core , -- I/O:
572  STREAM => stream , -- I/O:
573  KEY => "ON" , -- In :
574  VAL => wait_on , -- I/O:
575  EVENT => next_event -- I/O:
576  );
577  case next_event is
578  when EVENT_SCALAR =>
579  COPY_KEY_WORD(core, keyword);
580  EXECUTE_UNDEFINED_MAP_KEY(core, stream, keyword);
581  when EVENT_MAP_END =>
582  exit MAP_READ_LOOP;
583  when others =>
584  READ_ERROR(core, proc_name, "need EVENT_MAP_END but " &
585  EVENT_TO_STRING(next_event));
586  end case;
587  end loop;
588  if (wait_on) then
589  SIG_LOOP:loop
590  REPORT_DEBUG(core, proc_name, "SIG_LOOP");
591  wait_on_signals;
592  match_axi4_stream_signals(chk_signals, axi_match);
593  gpi_match := MATCH_STD_LOGIC(gpi_signals, GPI);
594  exit when(axi_match and gpi_match);
595  if (ACLK'event and ACLK = '1') then
596  if (timeout > 0) then
597  timeout := timeout - 1;
598  else
599  EXECUTE_ABORT(core, proc_name, "Time Out!");
600  end if;
601  end if;
602  end loop;
603  else
604  CLK_LOOP:loop
605  REPORT_DEBUG(core, proc_name, "CLK_LOOP");
606  wait until (ACLK'event and ACLK = '1');
607  match_axi4_stream_signals(chk_signals, axi_match);
608  gpi_match := MATCH_STD_LOGIC(gpi_signals, GPI);
609  exit when(axi_match and gpi_match);
610  if (timeout > 0) then
611  timeout := timeout - 1;
612  else
613  EXECUTE_ABORT(core, proc_name, "Time Out!");
614  end if;
615  end loop;
