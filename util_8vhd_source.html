300  procedure integer_to_std_logic_vector(
301  NUM : in integer;
302  VAL : out std_logic_vector
303  ) is
304  alias result : std_logic_vector(VAL'length-1 downto 0) is VAL;
305  variable number : integer;
306  variable bin : std_logic;
307  begin
308  if (NUM < 0) then
309  bin := '1';
310  number := -(NUM+1);
311  else
312  bin := '0';
313  number := NUM;
314  end if;
315  for i in 0 to result'left loop
316  if ((number mod 2) = 0) then
317  result(i) := bin;
318  else
319  result(i) := not bin;
320  end if;
321  number := number/2;
322  end loop;
323  end procedure;

332  procedure STRING_TO_STD_LOGIC_VECTOR(
333  STR : in string ;
334  VAL : out std_logic_vector;
335  STR_LEN : out integer ;
336  VAL_LEN : out integer
337  ) is
338  variable value : integer;
339  variable pos : integer;
340  variable len : integer;
341  begin
342  ---------------------------------------------------------------------------
343  -- 最初の文字が'0'の場合...
344  ---------------------------------------------------------------------------
345  if (STR(STR'low) = '0' and STR'low < STR'high) then
346  pos := STR'low;
347  if ((STR(pos+1) = 'x' or STR(pos+1) = 'X') and pos+2 <= STR'high) then
348  STRING_TO_HEX(STR(pos+2 to STR'high), VAL, len, VAL_LEN);
349  pos := pos + 2 + len;
350  else
351  STRING_TO_OCT(STR(pos+1 to STR'high), VAL, len, VAL_LEN);
352  pos := pos + 1 + len;
353  end if;
354  if (len > 0) then
355  STR_LEN := pos - STR'low;
356  else
357  STR_LEN := 0;
358  end if;
359  ---------------------------------------------------------------------------
360  -- 最初の文字が'0'ではなかった場合.
361  ---------------------------------------------------------------------------
362  else
363  -----------------------------------------------------------------------
364  -- 最初の定数(10進数)を読む.
365  -----------------------------------------------------------------------
366  STRING_TO_DEC(STR, value, len);
367  if (len = 0) then
368  value := 0;
369  end if;
370  pos := STR'low + len;
371  -----------------------------------------------------------------------
372  -- 次の字句が ' の場合は、最初に読んだ定数はベクタの大きさになる.
373  -----------------------------------------------------------------------
374  if (pos <= STR'high and STR(pos) = ''') then
375  if (value < 0 or pos+1 > STR'high) then
376  STR_LEN := 0;
377  VAL_LEN := 0;
378  return;
379  end if;
380  case STR(pos+1) is
381  when 'X'|'x'=> STRING_TO_HEX(STR(pos+2 to STR'high), VAL, len, VAL_LEN);
382  when 'H'|'h'=> STRING_TO_HEX(STR(pos+2 to STR'high), VAL, len, VAL_LEN);
383  when 'O'|'o'=> STRING_TO_OCT(STR(pos+2 to STR'high), VAL, len, VAL_LEN);
384  when 'B'|'b'=> STRING_TO_BIN(STR(pos+2 to STR'high), VAL, len, VAL_LEN);
385  when 'D'|'d'=> STRING_TO_DEC(STR(pos+2 to STR'high), VAL, len, VAL_LEN);
386  when others => len := 0; VAL_LEN := 0;
387  end case;
388  if (len > 0) then
389  if (value > 0) then
390  VAL_LEN := value;
391  end if;
392  STR_LEN := pos + 2 + len - STR'low;
393  else
394  STR_LEN := 0;
395  VAL_LEN := 0;
396  end if;
397  ---------------------------------------------------------------------------
398  -- 次の字句が ' でない場合は、読んだ定数を返す.
399  ---------------------------------------------------------------------------
400  else
401  STR_LEN := len;
402  VAL_LEN := VAL'length;
403  integer_to_std_logic_vector(value, VAL);
404  end if;
405  end if;
406  end procedure;
