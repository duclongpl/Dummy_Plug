460  procedure NEXT_PSEUDO_RANDOM_NUMBER_GENERATOR(
461  variable generator : inout PSEUDO_RANDOM_NUMBER_GENERATOR_TYPE
462  ) is
463  variable x : RANDOM_NUMBER_TYPE;
464  variable y : RANDOM_NUMBER_TYPE;
465  begin
466  x := (generator.status(0) and MASK) xor generator.status(1) xor generator.status(2);
467  y := generator.status(3);
468  x := x xor (x sll SH0);
469  y := y xor (y srl SH0) xor x;
470  generator.status(0) := generator.status(1);
471  generator.status(1) := generator.status(2);
472  generator.status(2) := x xor (y sll SH1);
473  generator.status(3) := y;
474  if (y(0) = '1') then
475  generator.status(1) := generator.status(1) xor generator.mat1;
476  generator.status(2) := generator.status(2) xor generator.mat2;
477  end if;
