154 package body SYNC is
155  -------------------------------------------------------------------------------
160  -------------------------------------------------------------------------------
161  procedure SYNC_BEGIN(
162  signal SYNC_REQ : out SYNC_REQ_VECTOR;
163  SYNC_CNT : in SYNC_REQ_VECTOR
164  ) is
165  begin
166  ---------------------------------------------------------------------------
167  -- SYNC_REQ信号に SYNC_CNT を出力.
168  ---------------------------------------------------------------------------
169  SYNC_REQ <= SYNC_CNT;
170  ---------------------------------------------------------------------------
171  -- SYNC_CNTの値がSYNC_REQ信号に反映されるまでのΔ時間だけ待つ.
172  ---------------------------------------------------------------------------
173  wait for 0 ns;
174  ---------------------------------------------------------------------------
175  -- SYNC_SIG_DRIVER にて SYNC_REQ信号が変化して SYNC 信号に反映されるまでの
176  -- Δ時間だけ待つ.
177  ---------------------------------------------------------------------------
178  wait for 0 ns;
179  ---------------------------------------------------------------------------
180  -- SYNC_SIG_DRIVER にて SYNC信号が変化して SYNC_ACK 信号に反映されるまで
181  -- のΔ時間だけ待つ.
182  ---------------------------------------------------------------------------
183  wait for 0 ns;
184  end SYNC_BEGIN;
185  -------------------------------------------------------------------------------
192  -------------------------------------------------------------------------------
193  procedure SYNC_CHECK(
194  signal SYNC_REQ : out SYNC_REQ_VECTOR;
195  signal SYNC_ACK : in SYNC_ACK_VECTOR;
196  SYNC_CNT : inout SYNC_REQ_VECTOR;
197  DONE : out boolean
198  ) is
199  variable use_sync : std_logic_vector(SYNC_CNT'range);
200  constant all_done : std_logic_vector(SYNC_CNT'range) := (others => '0');
201  begin
202  use_sync := (others => '1');
203  for i in SYNC_CNT'range loop
204  if (SYNC_CNT(i) > 0 and SYNC_ACK(i) = '1') then
205  SYNC_CNT(i) := 0;
206  SYNC_REQ(i) <= 0;
207  end if;
208  if (SYNC_CNT(i) > 0) then
209  use_sync(i) := '1';
210  else
211  use_sync(i) := '0';
212  end if;
213  end loop;
214  DONE := (use_sync = all_done);
215  end SYNC_CHECK;
216  -------------------------------------------------------------------------------
222  -------------------------------------------------------------------------------
223  procedure SYNC_END (
224  signal SYNC_REQ : out SYNC_REQ_VECTOR;
225  signal SYNC_ACK : in SYNC_ACK_VECTOR;
226  SYNC_CNT : in SYNC_REQ_VECTOR
227  ) is
228  variable count : SYNC_REQ_VECTOR (SYNC_CNT'range);
229  variable done : boolean;
230  begin
231  count := SYNC_CNT;
232  ---------------------------------------------------------------------------
233  -- SYNC_CHECKを実行して done が TRUE になるまで待つループ.
234  ---------------------------------------------------------------------------
235  WAIT_SYNC_ACK_LOOP : loop
236  SYNC_CHECK(SYNC_REQ => SYNC_REQ, -- Out: SYNC_REQ_VECTOR;
237  SYNC_ACK => SYNC_ACK, -- In : SYNC_ACK_VECTOR;
238  SYNC_CNT => count, -- I/O: SYNC_REQ_VECTOR;
239  DONE => done); -- Out: boolean;
240  exit when done;
241  wait on SYNC_ACK; -- 超重要 これがないとシミュレーターが無限ループに陥る
242  end loop;
243  ---------------------------------------------------------------------------
244  -- SYNC_CNTをクリアしたのがSYNC_REQ信号に反映されるまでのΔ時間だけ待つ.
245  ---------------------------------------------------------------------------
246  wait for 0 ns;
247  end SYNC_END;

275 architecture MODEL of SYNC_SIG_DRIVER is
276  procedure REPORT_DEBUG(MESSAGE: STRING) is
277  variable text_line : line;
278  begin
279  if (DEBUG) then
280  WRITE(text_line, Now, RIGHT, 13);
281  WRITE(text_line, string'(" (") & NAME & ") SYNC_SIG_DRIVER:");
282  WRITE(text_line, MESSAGE);
283  WRITELINE(OUTPUT, text_line);
284  end if;
285  end procedure;
286 begin
287  process
288  variable sync_delay : integer;
289  begin
290  SYNC <= 'Z';
291  ACK <= '0';
292  SYNC_LOOP: loop
293  if (REQ > 0) then
294  REPORT_DEBUG(string'("REQ>0"));
295  sync_delay := REQ;
296  if (sync_delay >= 2) then
297  for i in 2 to sync_delay loop
298  wait until (CLK'event and CLK = '1');
299  end loop;
300  end if;
301  SYNC <= 'H';
302  REPORT_DEBUG(string'("WAIT SYNC START"));
303  wait until (SYNC = '1' or SYNC = 'H');
304  REPORT_DEBUG(string'("WAIT SYNC DONE"));
305  SYNC <= '0';
306  ACK <= '1';
307  REPORT_DEBUG(string'("WAIT REQ START"));
308  wait until (REQ = 0);
309  REPORT_DEBUG(string'("WAIT REQ DONE"));
310  ACK <= '0';
311  elsif (REQ = 0) then
312  REPORT_DEBUG(string'("REQ=0"));
313  SYNC <= '0';
314  else
315  REPORT_DEBUG(string'("REQ<0"));
316  SYNC <= 'Z';
317  end if;
318  wait on REQ;
319  end loop;
320  end process;
321 end MODEL;
