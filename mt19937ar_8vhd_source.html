227  constant K : integer := 1812433253;

237  procedure init_by_array(
238  table : inout RANDOM_NUMBER_VECTOR;
239  init_key : in SEED_VECTOR
240  ) is
241  alias mt : RANDOM_NUMBER_VECTOR(0 to table'length-1) is table;
242  alias ik : SEED_VECTOR(0 to init_key'length-1) is init_key;
243  variable i : integer range mt'low to mt'high;
244  variable j : integer range ik'low to ik'high;
245  variable n : integer;
246  constant K1 : integer := 1664525;
247  constant K2 : integer := 1566083941;
248  begin
249  init_table(table, TO_RANDOM_NUMBER_TYPE(19650218));
250  i := 1;
251  j := 0;
252  if (mt'length > ik'length) then
253  n := mt'length;
254  else
255  n := ik'length;
256  end if;
257  for count in n downto 1 loop
258  mt(i) := (mt(i) xor MUL_K(K1,(mt(i-1) xor (mt(i-1) srl 30)))) + ik(j) + j;
259  if (i >= mt'high) then
260  mt(0) := mt(mt'high);
261  i := 1;
262  else
263  i := i + 1;
264  end if;
265  if (j >= ik'high) then
266  j := 0;
267  else
268  j := j + 1;
269  end if;
270  end loop;
271  for count in mt'high downto 1 loop
272  mt(i) := (mt(i) xor MUL_K(K2,(mt(i-1) xor (mt(i-1) srl 30)))) - i;
273  if (i >= mt'high) then
274  mt(0) := mt(mt'high);
275  i := 1;
276  else
277  i := i + 1;
278  end if;
279  end loop;

288  procedure generate_word(
289  variable generator : inout PSEUDO_RANDOM_NUMBER_GENERATOR_TYPE;
290  variable result : out RANDOM_NUMBER_TYPE
291  ) is
292  alias mt : RANDOM_NUMBER_VECTOR(0 to generator.table'length-1) is generator.table;
293  variable i : integer range mt'low to mt'high;
294  variable x0,x1,xm : RANDOM_NUMBER_TYPE;
295  variable y : RANDOM_NUMBER_TYPE;
296  variable z : RANDOM_NUMBER_TYPE;
297  constant mag01 : RANDOM_NUMBER_VECTOR(0 to 1) := (0 => X"00000000", 1 => MATRIX_A);
298  begin
299  i := generator.index;
300  x0 := mt(i);
301  x1 := mt((i+1) mod mt'length);
302  xm := mt((i+M) mod mt'length);
303  y := (x0 and UPPER_MASK) or (x1 and LOWER_MASK);
304  z := xm xor (y srl 1) xor mag01(to_integer(y mod mag01'length));
305  mt(i) := z;
306  generator.index := (i+1) mod mt'length;
307  y := z;
308  y := y xor ((y srl 11));
309  y := y xor ((y sll 7) and X"9d2c5680");
310  y := y xor ((y sll 15) and X"efc60000");
311  y := y xor ((y srl 18));
312  result := y;
313  end procedure;
