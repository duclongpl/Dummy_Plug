105 architecture MODEL of MARCHAL is
106  -------------------------------------------------------------------------------
108  -------------------------------------------------------------------------------
109  signal sync_req : SYNC_REQ_VECTOR(SYNC'range);
110  signal sync_ack : SYNC_ACK_VECTOR(SYNC'range);
111  signal sync_rst : std_logic := '0';
112  signal sync_clr : std_logic := '0';
113  signal sync_debug : boolean := FALSE;
114  -------------------------------------------------------------------------------
116  -------------------------------------------------------------------------------
117  subtype TIMER_TIME_TYPE is unsigned(63 downto 0);
118  constant DEFAULT_TIMEOUT : TIMER_TIME_TYPE := (others => '1');
119  signal timer_reset : boolean := TRUE;
120  signal timer_valid : boolean := TRUE;
121  signal timer : TIMER_TIME_TYPE := DEFAULT_TIMEOUT;
122  signal timeout_time : TIMER_TIME_TYPE := DEFAULT_TIMEOUT;
123 begin
124  -------------------------------------------------------------------------------
125  --
126  -------------------------------------------------------------------------------
127  MAIN: process
128  file stream : TEXT;
129  variable core : CORE_TYPE;
130  variable operation : OPERATION_TYPE;
131  variable keyword : STRING(1 to 3);
132  constant KEY_SAY: STRING(1 to 3) := "SAY";
133  constant KEY_DEBUG: STRING(1 to 3) := "DEB";
134  constant KEY_REPORT: STRING(1 to 3) := "REP";
135  constant KEY_SYNC: STRING(1 to 3) := "SYN";
136  constant KEY_WAIT: STRING(1 to 3) := "WAI";
137  constant KEY_TIMEOUT: STRING(1 to 3) := "TIM";
138  ---------------------------------------------------------------------------
140  ---------------------------------------------------------------------------
141  procedure EXECUTE_TIMEOUT is
142  constant PROC_NAME: string := "EXECUTE_TIMEOUT";
143  variable next_event : EVENT_TYPE;
144  variable valid : boolean;
145  variable count : std_logic_vector(TIMER_TIME_TYPE'range);
146  variable valid_len : integer;
147  variable count_len : integer;
148  begin
149  REPORT_DEBUG(core, PROC_NAME, "BEGIN");
150  SEEK_EVENT(core, stream, next_event);
151  case next_event is
152  when EVENT_SCALAR =>
153  READ_EVENT(core, stream, EVENT_SCALAR);
154  STRING_TO_STD_LOGIC_VECTOR(core.str_buf(1 to core.str_len), count, count_len);
155  if (count_len > 0) then
156  timeout_time <= unsigned(count);
157  else
158  STRING_TO_BOOLEAN (core.str_buf(1 to core.str_len), valid, valid_len);
159  if (valid_len > 0) then
160  timer_valid <= valid;
161  else
162  READ_ERROR(core, PROC_NAME, "Invalid Parameter(" &
163  core.str_buf(1 to core.str_len) & ")");
164  end if;
165  end if;
166  when others =>
167  READ_ERROR(core, PROC_NAME, "SEEK_EVENT NG");
168  end case;
169  REPORT_DEBUG(core, PROC_NAME, "END");
170  end procedure;
171  ---------------------------------------------------------------------------
173  ---------------------------------------------------------------------------
174  procedure EXECUTE_WAIT is
175  constant PROC_NAME : string := "EXECUTE_WAIT";
176  variable next_event : EVENT_TYPE;
177  variable wait_count : integer;
178  variable scan_len : integer;
179  begin
180  REPORT_DEBUG(core, PROC_NAME, "BEGIN");
181  SEEK_EVENT(core, stream, next_event);
182  case next_event is
183  when EVENT_SCALAR =>
184  READ_EVENT(core, stream, EVENT_SCALAR);
185  STRING_TO_INTEGER(core.str_buf(1 to core.str_len), wait_count, scan_len);
186  if (scan_len = 0) then
187  wait_count := 1;
188  end if;
189  if (wait_count > 0) then
190  for i in 1 to wait_count loop
191  wait until (CLK'event and CLK = '1');
192  end loop;
193  end if;
194  wait_count := 0;
195  when others =>
196  READ_ERROR(core, PROC_NAME, "SEEK_EVENT NG");
197  end case;
198  REPORT_DEBUG(core, PROC_NAME, "END");
199  end procedure;
200  ---------------------------------------------------------------------------
202  ---------------------------------------------------------------------------
203  procedure EXECUTE_SYNC(operation: in OPERATION_TYPE) is
204  constant PROC_NAME : string := "EXECUTE_SYNC";
205  variable port_num : integer;
206  variable wait_num : integer;
207  begin
208  REPORT_DEBUG (core, PROC_NAME, "BEGIN");
209  READ_SYNC_ARGS(core, stream, operation, port_num, wait_num);
210  REPORT_DEBUG (core, PROC_NAME, "PORT=" & INTEGER_TO_STRING(port_num) &
211  " WAIT=" & INTEGER_TO_STRING(wait_num));
212  if (SYNC_REQ'low <= port_num and port_num <= SYNC_REQ'high) then
213  CORE_SYNC(core, port_num, wait_num, SYNC_REQ, SYNC_ACK);
214  end if;
215  REPORT_DEBUG (core, PROC_NAME, "END");
216  end procedure;
217  begin
218  ---------------------------------------------------------------------------
220  ---------------------------------------------------------------------------
221  CORE_INIT(
222  SELF => core,
223  NAME => NAME,
224  STREAM => stream,
225  STREAM_NAME => SCENARIO_FILE,
226  OPERATION => operation
227  );
228  ---------------------------------------------------------------------------
230  ---------------------------------------------------------------------------
231  REPORT_STATUS <= core.report_status;
232  sync_req <= (0 => 10, others => 0);
233  FINISH <= '0';
234  sync_rst <= '1';
235  sync_clr <= '0';
236  RESET <= '1';
237  wait until(CLK'event and CLK = '1');
238  sync_rst <= '0';
239  RESET <= '0';
240  wait until(CLK'event and CLK = '1');
241  ---------------------------------------------------------------------------
243  ---------------------------------------------------------------------------
244  core.debug := 0;
245  MAIN_LOOP: while (operation /= OP_FINISH) loop
246  REPORT_STATUS <= core.report_status;
247  READ_OPERATION(core, stream, operation, keyword);
248  case operation is
249  when OP_DOC_BEGIN =>
250  timer_reset <= TRUE, FALSE after 1 ns;
251  CORE_SYNC(core, 0, 2, sync_req, sync_ack);
252  when OP_MAP =>
253  case keyword is
254  when KEY_SYNC => EXECUTE_SYNC (operation);
255  when KEY_WAIT => EXECUTE_WAIT;
256  when KEY_TIMEOUT=> EXECUTE_TIMEOUT;
257  when KEY_SAY => EXECUTE_SAY (core, stream);
258  when KEY_DEBUG => EXECUTE_DEBUG (core, stream);
259  when KEY_REPORT => EXECUTE_REPORT(core, stream);
260  when others => EXECUTE_SKIP (core, stream);
261  end case;
262  when OP_SCALAR =>
263  case keyword is
264  when KEY_SYNC => EXECUTE_SYNC (operation);
265  when others => null;
266  end case;
267  when OP_FINISH => exit;
268  when others => null;
269  end case;
270  end loop;
271  REPORT_STATUS <= core.report_status;
272  FINISH <= '1';
273  if (FINISH_ABORT) then
274  assert FALSE report "Simulation complete." severity FAILURE;
275  end if;
276  wait;
277  end process;
278  -------------------------------------------------------------------------------
280  -------------------------------------------------------------------------------
281  TIME_KEEPER: process(timer_reset, CLK)
282  variable vocal : VOCAL_TYPE := NEW_VOCAL(NAME);
283  begin
284  if (timer_reset) then
285  timer <= (others => '0');
286  elsif (CLK'event and CLK = '1') then
287  if (timer_valid) then
288  if (timer < timeout_time) then
289  timer <= timer + 1;
290  else
291  REPORT_FAILURE(vocal, string'("Time Out!"));
292  assert FALSE report "Time Out!" severity FAILURE;
293  end if;
294  end if;
295  end if;
296  end process;

299  -------------------------------------------------------------------------------

302  generic map (
303  NAME => string'("MARCHAL:SYNC"),
304  PLUG_NUM => SYNC_PLUG_NUM
305  )
306  port map (
307  CLK => CLK , -- In :
308  RST => sync_rst , -- In :
309  CLR => sync_clr , -- In :
310  DEBUG => sync_debug, -- In :
311  SYNC => SYNC(i), -- I/O:
312  REQ => sync_req (i), -- In :
313  ACK => sync_ack (i) -- Out:
314  );
315  end generate;
316 end MODEL;
