874  procedure check_my_name(
875  variable SELF : inout CORE_TYPE;
876  file STREAM : TEXT;
877  variable FOUND : out boolean
878  ) is
879  constant PROC_NAME : string := "CORE.check_my_name";
880  variable get_event : EVENT_TYPE;
881  variable seq_level : integer;
882  variable match : boolean;
883  begin
884  seq_level := 0;
885  FOUND := FALSE;
886  MAIN_LOOP: loop
887  SEEK_EVENT(SELF, STREAM, get_event);
888  case get_event is
889  when EVENT_SEQ_BEGIN =>
890  READ_EVENT(SELF, STREAM, get_event);
891  seq_level := seq_level + 1;
892  when EVENT_SEQ_END =>
893  if (seq_level > 0) then
894  READ_EVENT(SELF, STREAM, get_event);
895  seq_level := seq_level - 1;
896  end if;
897  exit when (seq_level = 0);
898  when EVENT_SCALAR =>
899  READ_EVENT(SELF, STREAM, get_event);
900  MATCH_KEY_WORD(SELF, SELF.name(SELF.name'range), match);
901  if (match) then
902  FOUND := TRUE;
903  end if;
904  exit when (seq_level = 0);
905  when EVENT_ERROR =>
906  READ_ERROR(SELF, PROC_NAME, "SEEK_EVENT NG");
907  when others =>
908  SKIP_EVENT(SELF, STREAM, get_event);
909  end case;
910  end loop;
911  end procedure;

920  procedure check_first_node(
921  variable SELF : inout CORE_TYPE;
922  file STREAM : TEXT;
923  variable NEXT_STATE : out STATE_TYPE
924  ) is
925  constant PROC_NAME : string := "CORE.check_first_node";
926  variable next_event : EVENT_TYPE;
927  variable found : boolean;
928  begin
929  CHECK_LOOP: loop
930  SEEK_EVENT(SELF, STREAM, next_event);
931  case next_event is
932  -------------------------------------------------------------------
933  -- エラーだった場合.
934  -------------------------------------------------------------------
935  when EVENT_ERROR =>
936  READ_ERROR(SELF, PROC_NAME, "SEEK_EVENT NG");
937  -------------------------------------------------------------------
938  -- 最初のノードがマップだった場合.
939  -- * 例１ ID: {...}
940  -- * 例２ ID: [{...},{...}]
941  -------------------------------------------------------------------
942  when EVENT_MAP_BEGIN =>
943  READ_EVENT(SELF, STREAM, next_event);
944  check_my_name(SELF, STREAM, found);
945  if (found) then
946  SEEK_EVENT(SELF, STREAM, next_event);
947  if (next_event = EVENT_SEQ_BEGIN) then
948  READ_EVENT(SELF, STREAM, next_event);
949  NEXT_STATE := STATE_MAP_SEQ;
950  else
951  NEXT_STATE := STATE_MAP_VAL;
952  end if;
953  else
954  SEEK_EVENT(SELF, STREAM, next_event);
955  SKIP_EVENT(SELF, STREAM, next_event);
956  NEXT_STATE := STATE_TOP_SEQ;
957  end if;
958  exit;
959  -------------------------------------------------------------------
960  -- 最初のノードがシーケンスだった場合.
961  -- * 例１ [ID, {...}]
962  -------------------------------------------------------------------
963  when EVENT_SEQ_BEGIN =>
964  READ_EVENT(SELF, STREAM, next_event);
965  check_my_name(SELF, STREAM, found);
966  if (found) then
967  NEXT_STATE := STATE_SEQ_VAL;
968  else
969  NEXT_STATE := STATE_SEQ_SKIP;
970  end if;
971  exit;
972  -------------------------------------------------------------------
973  -- 最初のノードがTAG PROPERTYまたはアンカーの場合は単純に読み飛ばす.
974  -------------------------------------------------------------------
975  when EVENT_TAG_PROP | EVENT_ANCHOR =>
976  SKIP_EVENT(SELF, STREAM, next_event);
977  next;
978  -------------------------------------------------------------------
979  -- 最初のノードがマップでもシーケンスでもなかった場合はエラー.
980  -------------------------------------------------------------------
981  when others =>
982  SKIP_EVENT(SELF, STREAM, next_event);
983  NEXT_STATE := STATE_TOP_SEQ;
984  exit;
985  end case;
986  end loop;
987  end procedure;

994  procedure COPY_KEY_WORD(
995  variable SELF : inout CORE_TYPE;
996  KEY_WORD : out STRING
997  ) is
998  variable skip : boolean;
999  alias key_buf : string(1 to KEY_WORD'length) is KEY_WORD;
1000  begin

1051  procedure READ_OPERATION(
1052  variable SELF : inout CORE_TYPE;
1053  file STREAM : TEXT;
1054  variable OPERATION : out OPERATION_TYPE;
1055  variable OP_WORD : out string
1056  ) is
1057  constant PROC_NAME : string := "CORE.READ_OPERATION";
1058  variable next_event : EVENT_TYPE;
1059  variable next_state : STATE_TYPE;
1060  procedure REPORT_DEBUG(state:in STRING;event:EVENT_TYPE) is
1061  begin
1062  REPORT_DEBUG(SELF, PROC_NAME, string'("state=") & state &
1063  string'(" next_event=") & EVENT_TO_STRING(event));
1064  if (SELF.debug > 1) then
1065  DEBUG_DUMP(SELF.reader);
1066  end if;
1067  end procedure;
1068  begin
1069  SELF.vocal.enable_debug := TRUE;
1070  MAIN_LOOP: loop
1071  SEEK_EVENT(SELF, STREAM, next_event);
1072  if (next_event = EVENT_ERROR) then
1073  READ_ERROR(SELF, PROC_NAME, "SEEK_EVENT NG");
1074  end if;
1075  case SELF.curr_state is
1076  -------------------------------------------------------------------
1077  -- 初期状態.
1078  -------------------------------------------------------------------
1079  when STATE_NULL | STATE_STREAM =>
1080  REPORT_DEBUG(string'("STATE_STREAM"),next_event);
1081  case next_event is
1082  when EVENT_DOC_BEGIN =>
1083  READ_EVENT(SELF, STREAM, next_event);
1084  SELF.curr_state := STATE_DOCUMENT;
1085  OPERATION := OP_DOC_BEGIN;
1086  exit;
1087  when EVENT_STREAM_END =>
1088  READ_EVENT(SELF, STREAM, next_event);
1089  SELF.curr_state := STATE_NULL;
1090  OPERATION := OP_FINISH;
1091  exit;
1092  when others =>
1093  SKIP_EVENT(SELF, STREAM, next_event);
1094  end case;
1095  -------------------------------------------------------------------
1096  -- ドキュメント処理中.
1097  -------------------------------------------------------------------
1098  when STATE_DOCUMENT =>
1099  REPORT_DEBUG(string'("STATE_DOCUMENT"), next_event);
1100  case next_event is
1101  when EVENT_SEQ_BEGIN =>
1102  READ_EVENT(SELF, STREAM, next_event);
1103  SELF.curr_state := STATE_TOP_SEQ;
1104  when EVENT_DOC_END =>
1105  READ_EVENT(SELF, STREAM, next_event);
1106  SELF.curr_state := STATE_STREAM;
1107  OPERATION := OP_DOC_END;
1108  exit;
1109  when others =>
1110  READ_EVENT(SELF, STREAM, next_event);
1111  -- ERROR
1112  end case;
1113  -------------------------------------------------------------------
1114  -- 最も外側のシーケンス処理中.
1115  -- シーケンスの各ノードに現れる最初のノードによって、
1116  -- 以降の状態が変化する.
1117  -- * 例１ - ID:{...} #=>STATE_MAP_VAL
1118  -- * 例２ - ID:[{...},{...}] #=>STATE_MAP_SEQ
1119  -- * 例３ - [ID,{...},{...}] #=>STATE_SEQ_VAL or STATE_SEQ_SKIP
1120  -------------------------------------------------------------------
1121  when STATE_TOP_SEQ =>
1122  REPORT_DEBUG(string'("STATE_TOP_SEQ"), next_event);
1123  case next_event is
1124  when EVENT_SEQ_END =>
1125  READ_EVENT(SELF, STREAM, next_event);
1126  SELF.curr_state := STATE_DOCUMENT;
1127  when others =>
1128  check_first_node(SELF, STREAM, next_state);
1129  SELF.curr_state := next_state;
1130  end case;
1131  -------------------------------------------------------------------
1132  -- ID:{...} を処理する状態.
1133  -------------------------------------------------------------------
1134  when STATE_MAP_VAL =>
1135  REPORT_DEBUG(string'("STATE_MAP_VAL"), next_event);
1136  case next_event is
1137  when EVENT_MAP_BEGIN =>
1138  READ_EVENT(SELF, STREAM, next_event);
1139  SEEK_EVENT(SELF, STREAM, next_event);
1140  READ_EVENT(SELF, STREAM, next_event);
1141  COPY_KEY_WORD(SELF, OP_WORD);
1142  SELF.curr_state := STATE_OP_MAP;
1143  SELF.prev_state := STATE_MAP_END;
1144  OPERATION := OP_MAP;
1145  exit;
1146  when EVENT_SCALAR =>
1147  READ_EVENT(SELF, STREAM, next_event);
1148  COPY_KEY_WORD(SELF, OP_WORD);
1149  SELF.curr_state := STATE_OP_SCALAR;
1150  SELF.prev_state := STATE_TOP_SEQ;
1151  OPERATION := OP_SCALAR;
1152  exit;
1153  when others =>
1154  SKIP_EVENT(SELF, STREAM, next_event);
1155  end case;
1156  -------------------------------------------------------------------
1157  -- ID:[{...},{...}] を処理する状態.
1158  -------------------------------------------------------------------
1159  when STATE_MAP_SEQ =>
1160  REPORT_DEBUG(string'("STATE_MAP_SEQ"), next_event);
1161  case next_event is
1162  when EVENT_SEQ_END =>
1163  READ_EVENT(SELF, STREAM, next_event);
1164  SELF.curr_state := STATE_MAP_END;
1165  when EVENT_MAP_BEGIN =>
1166  READ_EVENT(SELF, STREAM, next_event);
1167  SEEK_EVENT(SELF, STREAM, next_event);
1168  READ_EVENT(SELF, STREAM, next_event);
1169  COPY_KEY_WORD(SELF, OP_WORD);
1170  SELF.curr_state := STATE_OP_MAP;
1171  SELF.prev_state := STATE_MAP_SEQ;
1172  OPERATION := OP_MAP;
1173  exit;
1174  when EVENT_SCALAR =>
1175  READ_EVENT(SELF, STREAM, next_event);
1176  COPY_KEY_WORD(SELF, OP_WORD);
1177  SELF.curr_state := STATE_OP_SCALAR;
1178  SELF.prev_state := STATE_MAP_SEQ;
1179  OPERATION := OP_SCALAR;
1180  exit;
1181  when others =>
1182  SKIP_EVENT(SELF, STREAM, next_event);
1183  end case;
1184  -------------------------------------------------------------------
1185  -- [ID,{...},{...}] を処理する状態.
1186  -------------------------------------------------------------------
1187  when STATE_SEQ_VAL =>
1188  REPORT_DEBUG(string'("STATE_SEQ_VAL"), next_event);
1189  case next_event is
1190  when EVENT_SEQ_END =>
1191  READ_EVENT(SELF, STREAM, next_event);
1192  SELF.curr_state := STATE_TOP_SEQ;
1193  when EVENT_MAP_BEGIN =>
1194  READ_EVENT(SELF, STREAM, next_event);
1195  SEEK_EVENT(SELF, STREAM, next_event);
1196  READ_EVENT(SELF, STREAM, next_event);
1197  COPY_KEY_WORD(SELF, OP_WORD);
1198  SELF.curr_state := STATE_OP_MAP;
1199  SELF.prev_state := STATE_SEQ_VAL;
1200  OPERATION := OP_MAP;
1201  exit;
1202  when EVENT_SCALAR =>
1203  READ_EVENT(SELF, STREAM, next_event);
1204  COPY_KEY_WORD(SELF, OP_WORD);
1205  SELF.curr_state := STATE_OP_SCALAR;
1206  SELF.prev_state := STATE_SEQ_VAL;
1207  OPERATION := OP_SCALAR;
1208  exit;
1209  when others =>
1210  SKIP_EVENT(SELF, STREAM, next_event);
1211  end case;
1212  -------------------------------------------------------------------
1213  -- [ID,{...},{...}]を読み飛ばす状態.
1214  -------------------------------------------------------------------
1215  when STATE_SEQ_SKIP =>
1216  REPORT_DEBUG(string'("STATE_SEQ_SKIP"), next_event);
1217  case next_event is
1218  when EVENT_SEQ_END =>
1219  READ_EVENT(SELF, STREAM, next_event);
1220  SELF.curr_state := STATE_TOP_SEQ;
1221  when others =>
1222  SKIP_EVENT(SELF, STREAM, next_event);
1223  end case;
1224  -------------------------------------------------------------------
1225  -- {ID: {...}} または {ID: [{...},{...}]}の最後のEVENT_MAP_ENDを
1226  -- 読む状態.
1227  -------------------------------------------------------------------
1228  when STATE_MAP_END =>
1229  REPORT_DEBUG(string'("STATE_MAP_END"), next_event);
1230  case next_event is
1231  when EVENT_MAP_END =>
1232  READ_EVENT(SELF, STREAM, next_event);
1233  SELF.curr_state := STATE_TOP_SEQ;
1234  when others =>
1235  READ_ERROR(SELF, PROC_NAME,
1236  "need EVENT_MAP_END but " &
1237  EVENT_TO_STRING(next_event) &
1238  " in STATE_MAP_END");
1239  end case;
1240  -------------------------------------------------------------------
1241  -- OP_MAPを処理している状態.
1242  -------------------------------------------------------------------
1243  when STATE_OP_MAP =>
1244  REPORT_DEBUG(string'("STATE_OP_MAP"), next_event);
1245  case next_event is
1246  when EVENT_MAP_END =>
1247  READ_EVENT(SELF, STREAM, next_event);
1248  SELF.curr_state := SELF.prev_state;
1249  when others =>
1250  READ_ERROR(SELF, PROC_NAME,
1251  "need EVENT_MAP_END but " &
1252  EVENT_TO_STRING(next_event) &
1253  " in STATE_OP_MAP");
1254  end case;
1255  -------------------------------------------------------------------
1256  -- OP_SCALARを処理している状態.
1257  -------------------------------------------------------------------
1258  when STATE_OP_SCALAR =>
1259  REPORT_DEBUG(string'("STATE_OP_SCALAR"), next_event);
1260  SELF.curr_state := SELF.prev_state;
1261  -------------------------------------------------------------------
1262  -- 不正な状態(ありえないはず).
1263  -------------------------------------------------------------------
1264  when others =>
1265  EXECUTE_ABORT(SELF, PROC_NAME, "bad state");
1266  end case;
1267  end loop;
1268  end procedure;

1278  procedure READ_SYNC_ARGS(
1279  variable SELF : inout CORE_TYPE;
1280  file STREAM : TEXT;
1281  OPERATION : in OPERATION_TYPE;
1282  SYNC_PORT : out integer;
1283  SYNC_WAIT : out integer
1284  ) is
1285  constant PROC_NAME : string := "CORE.READ_SYNC_ARGS";
1286  type STATE_TYPE is (
1287  STATE_NULL,
1288  STATE_SCALAR_PORT,
1289  STATE_MAP_KEY,
1290  STATE_MAP_PORT,
1291  STATE_MAP_WAIT,
1292  STATE_ERROR
1293  );
1294  variable state : STATE_TYPE;
1295  variable next_event : EVENT_TYPE;
1296  variable port_num : integer;
1297  variable wait_num : integer;
1298  variable scan_len : integer;
1299  variable match : boolean;
1300  variable map_level : integer;
1301  variable keyword : STRING(1 to 5);
1302  constant KEY_WAIT : STRING(1 to 5) := "WAIT ";
1303  constant KEY_PORT : STRING(1 to 5) := "PORT ";
1304  constant KEY_LOCAL : STRING(1 to 5) := "LOCAL";
1305  begin
1306  port_num := 0;
1307  wait_num := 2;
1308  case OPERATION is
1309  when OP_MAP =>
1310  map_level := 0;
1311  state := STATE_SCALAR_PORT;
1312  OP_MAP_LOOP: loop
1313  SEEK_EVENT(SELF, STREAM, next_event);
1314  case next_event is
1315  when EVENT_MAP_BEGIN =>
1316  READ_EVENT(SELF, STREAM, next_event);
1317  map_level := map_level + 1;
1318  state := STATE_MAP_KEY;
1319  when EVENT_MAP_END =>
1320  READ_EVENT(SELF, STREAM, next_event);
1321  map_level := map_level - 1;
1322  state := STATE_NULL;
1323  when EVENT_SCALAR =>
1324  READ_EVENT(SELF, STREAM, next_event);
1325  case state is
1326  when STATE_MAP_KEY =>
1327  COPY_KEY_WORD(SELF, keyword);
1328  case keyword is
1329  when KEY_PORT => state := STATE_MAP_PORT;
1330  when KEY_WAIT => state := STATE_MAP_WAIT;
1331  when others => state := STATE_ERROR;
1332  end case;
1333  when STATE_SCALAR_PORT | STATE_MAP_PORT =>
1334  COPY_KEY_WORD(SELF, keyword);
1335  if (keyword = KEY_LOCAL) then
1336  port_num := -1;
1337  else
1338  STRING_TO_INTEGER(SELF.str_buf(1 to SELF.str_len), port_num, scan_len);
1339  end if;
1340  if (state = STATE_MAP_PORT) then
1341  state := STATE_MAP_KEY;
1342  else
1343  state := STATE_NULL;
1344  end if;
1345  when STATE_MAP_WAIT =>
1346  STRING_TO_INTEGER(SELF.str_buf(1 to SELF.str_len), wait_num, scan_len);
1347  state := STATE_MAP_KEY;
1348  when others =>
1349  state := STATE_MAP_KEY;
1350  end case;
1351  when EVENT_ERROR =>
1352  READ_ERROR(SELF, PROC_NAME, "SEEK_EVENT NG");
1353  when others =>
1354  SKIP_EVENT(SELF, stream, next_event);
1355  end case;
1356  exit when (map_level = 0);
1357  end loop;
1358  if (next_event /= EVENT_MAP_END) then
1359  READ_ERROR(SELF, PROC_NAME, "need EVENT_MAP_END but " &
1360  EVENT_TO_STRING(next_event));
1361  end if;
1362  when OP_DOC_BEGIN => null;
1363  when OP_SCALAR => null;
1364  when others => null;
1365  end case;
1366  SYNC_PORT := port_num;
1367  SYNC_WAIT := wait_num;
1368  end procedure;

1536  constant KEY_WARNING : STRING(1 to 3) := "WAR";

1539  constant KEY_FAILURE : STRING(1 to 3) := "FAI";

1572  when KEY_ERROR => null;
1573  when KEY_FAILURE => null;
